const gridContainer = document.querySelector('#grid-container');
const buttonContainer = document.querySelector('#button-div');
const buttonBlue = document.createElement('button');


function createGridDivs(col, row) {
    for (let i = 0; i < (col * row); i++) {
        const div = document.createElement('div')
        div.classList.add('div')
    gridContainer.style.gridTemplateColumns = `repeat(${col} ifr)`;
    gridContainer.style.gridTemplateRows = `repeat(${row} ifr)`;
    gridContainer.appendChild(div);
    }
}
createGridDivs(16, 16)



function changeToBlue(){
const divs = gridContainer.querySelectorAll('.div'); 
buttonBlue.textContent = 'blueify';
buttonBlue.addEventListener('click' , () => {
    divs.forEach(div => div.addEventListener('mouseover', () => {
        div.style.cssText = 'background: red';
    }))
})
buttonContainer.appendChild(buttonBlue).classList.add('btn')
}

changeToBlue();
